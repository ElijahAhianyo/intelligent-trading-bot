def load_config(config_file):
    if config_file:
        config_file_path = PACKAGE_ROOT / config_file
        with open(config_file_path, encoding='utf-8') as json_file:
            conf_str = json_file.read()
            conf_str = re.sub('//.*$', '', conf_str, flags=re.M)
